(defn prime? [n]
  (= n (smallest-divisor n)))

(defn search-for-primes [n]
  (time (take 3 (all-primes-starting-at n))))

(defn all-primes-starting-at [n]
  (filter #(not (nil? %))
    (map #(when (prime? %) %) (iterate inc n))))
