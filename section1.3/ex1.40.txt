(defn deriv [g]
  (let [dx 0.00001]
    #(/ (- (g (+ % dx)) (g %)) dx)))

(defn newton-transform [g]
  #(- % (/ (g %) ((deriv g) %))))

(defn newtons-method [g guess]
  (fixed-point (newton-transform g) guess))

(defn cubic [a b c]
  #(+ (* % % %) (* a % %) (* b %) c))
